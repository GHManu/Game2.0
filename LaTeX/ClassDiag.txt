@startuml
skinparam class {
    BackgroundColor White
    ArrowColor Black
    BorderColor Black
    ' Colori personalizzati per i cerchi (opzionale)
    BackgroundColor<<interface>> LighBlue
    BackgroundColor<<abstract>> Wheat
}


' --- CLASSI PRINCIPALI ---

class GameUpdate {
    +run() : void
    +updateMovement(dt:double):void
    +updateAttack(dt):void
    +setState(state:IGameLoopState):void
    +getDestroyer(): Destroyer
}

interface IGameLoopState <<interface>> {
    {abstract} +update(context:GameUpdate, dt)
}

class PlayingState {
    +update(context, dt)
}

' --- GESTORI (HANDLERS) ---

class MovementHandler {
    +handle(dt, characters:List<ACharacter>,
    enemy:ACharacterEnemy,
    player:ACharacterPlayable)
}

class CollisionHandler {
    +handle(characters)
}

class AttackHandler {
    +handle(dt, player, enemy, characters)
}

' --- PERSONAGGI ---

abstract class ACharacter <<abstract>> {
    {abstract} +isAlive(): boolean
}

class ACharacterPlayable {
    +setInit_attack_flag(bool)
    +selectAttack(dt, player, enemy, characters)
}

class ACharacterEnemy {
    +selectAttack(dt, player, enemy)
}

' --- STRATEGIE DI ATTACCO (Strategy Pattern) ---

interface IFightStrategyPlayer <<interface>> {
    {abstract} +normalAttack(dt, player, characters)
}

interface IFightStrategyEnemy <<interface>> {
    {abstract} +normalAttack(dt, enemy, player):void
}

abstract class ACommonAttack <<abstract>> {
    #applyDamage(target:ACharacter,
    damage:double):void
}

class CommonAttackFireWeaponPlayer {
    +normalAttack(dt, player, characters)
}

class CommonAttackFireWeaponEnemy {
    +normalAttack(dt, enemy, player)
}

' --- UTILITY E UI ---

class ProjectileCollisionResolver {
    +hitWall(projectile,
    walls:List<Collider>): boolean
    +hitEnemy(projectile:AProjectile,
    characters): ACharacterEnemy
    +hitEnemyProjectile(projectile, 
    weapon:AFireWeapon): boolean
}

class HUD {
    +updateBars(character)
}

' --- RELAZIONI ---

' Gerarchie
IGameLoopState <|.. PlayingState
ACharacter <|-- ACharacterPlayable
ACharacter <|-- ACharacterEnemy

ACommonAttack <|-- CommonAttackFireWeaponPlayer
ACommonAttack <|-- CommonAttackFireWeaponEnemy
IFightStrategyPlayer <|.. CommonAttackFireWeaponPlayer
IFightStrategyEnemy <|.. CommonAttackFireWeaponEnemy

' Associazioni e Dipendenze
GameUpdate o-- IGameLoopState
GameUpdate --> MovementHandler
GameUpdate --> AttackHandler
GameUpdate --> HUD

PlayingState ..> GameUpdate : "cambia stato / invoca update"

MovementHandler --> CollisionHandler

AttackHandler --> ACharacterPlayable : "imposta flag"
ACharacterPlayable --> IFightStrategyPlayer : "delega attacco"
ACharacterEnemy --> IFightStrategyEnemy : "delega attacco"

CommonAttackFireWeaponPlayer --> ProjectileCollisionResolver : "usa per calcolo"
CommonAttackFireWeaponPlayer ..> ACharacterEnemy : "applica danno"

CommonAttackFireWeaponEnemy ..> ACharacterPlayable : "applica danno"

@enduml
